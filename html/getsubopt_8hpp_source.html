42  strings in TOKENS. If found return index and set *VALUEP to
43  optional value introduced by an equal sign. If the suboption is
44  not part of TOKENS return in *VALUEP beginning of unknown
45  suboption. On exit *OPTIONP is set to the beginning of the next
46  token or at the terminating NUL character. */

62  if (vstart == NULL)

67  for (cnt = 0; tokens[cnt] != NULL; ++cnt)

69  && tokens[cnt][vstart - *optionp] == '\0')
70  {
71  /* We found the current option in TOKENS. */
