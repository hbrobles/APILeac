662 const char * gnuplot_tmpfile(gnuplot_ctrl * handle)
663 {
664 #if defined(_WIN32) || defined(_WIN64)
665  static const char * tmp_filename_template = "gnuplot_tmpdatafile_XXXXXX";
666 #else
667  int unx_fd;
668  static const char * tmp_filename_template = "/tmp/gnuplot_tmpdatafile_XXXXXX";
669 #endif //#if defined(_WIN32) || defined(_WIN64)
670  char * tmp_filename = NULL;
671  int tmp_filelen = strlen(tmp_filename_template);
672 
673  assert(handle->tmp_filename_tbl[handle->ntmp] == NULL);
674 
675  /* Open one more temporary file? */
676  if (handle->ntmp == GP_MAX_TMP_FILES - 1) {
677  fprintf(stderr,
678  "maximum # of temporary files reached (%d): cannot open more",
679  GP_MAX_TMP_FILES) ;
680  return NULL;
681  }
682 
683  tmp_filename = (char*) malloc(tmp_filelen+1);
684  if (tmp_filename == NULL)
685  {
686  return NULL;
687  }
688  strcpy(tmp_filename, tmp_filename_template);
689 
690 #if defined(_WIN32) || defined(_WIN64)
691  if (_mktemp(tmp_filename) == NULL)
692  {
693  return NULL;
694  }
695 #else // #ifdef WIN32
696  unx_fd = mkstemp(tmp_filename);
697  if (unx_fd == -1)
698  {
699  return NULL;
700  }
701  close(unx_fd);
702 
703 #endif //#if defined(_WIN32) || defined(_WIN64)
704 
705  handle->tmp_filename_tbl[handle->ntmp] = tmp_filename;
706  handle->ntmp ++;
707 
708  return tmp_filename;
709 }
